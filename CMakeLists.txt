cmake_minimum_required (VERSION 3.0)
set(CMAKE_CXX_COMPILER /usr/bin/g++-8)
project (lqqcmmtools)
enable_language (CXX)
include_directories (./include)
file (GLOB SOURCES "src/*.cpp")
add_executable(lqqcmmtools ${SOURCES})
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS}  -O3  -std=c++17 -lstdc++fs ")
target_link_libraries(lqqcmmtools -lstdc++fs)
